#瓦斯的浓度在百分之5到百分之16之间会导致什么结果？
MATCH(:瓦斯)-[:浓度]->(n)-[:导致]->(q)
WHERE n.name='百分之5到百分之16'
RETURN '瓦斯','浓度',n.name,'导致',q.name

#爆炸的原因是什么？
MATCH (:爆炸)-[:原因]->(n)
RETURN '爆炸','原因',n.name

#通风积聚的防范措施有哪些？
MATCH (:通风积聚)-[:防范措施]->(n)
RETURN '通风积聚','防范措施',n.name

#出现明火或火花的防范措施有哪些？
MATCH (:出现明火或火花)-[:防范措施]->(n)
RETURN '出现明火或火花','防范措施',n.name

#不同的瓦斯浓度会导致什么事件？
MATCH (:瓦斯)-[:浓度]->(a)-[:导致]->(b)
RETURN '瓦斯','浓度',a.name,'导致',b.name
